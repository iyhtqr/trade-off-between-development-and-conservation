    <script>

        // --- 1. SETUP ---

        const scene = new THREE.Scene();

        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);

        camera.position.set(15, 15, 15);


        const renderer = new THREE.WebGLRenderer({ antialias: true });

        renderer.setSize(window.innerWidth, window.innerHeight);

        renderer.shadowMap.enabled = true;

        document.body.appendChild(renderer.domElement);


        const world = new CANNON.World();

        world.gravity.set(0, -50, 0);

        world.defaultContactMaterial.friction = 1.0;

        world.defaultContactMaterial.restitution = 0.0;

        world.defaultContactMaterial.contactEquationStiffness = 1e8;

        world.defaultContactMaterial.contactEquationRelaxation = 3;


        // Materials

        const groundMat = new CANNON.Material();

        const devMat = new CANNON.Material();

        const heritageMat = new CANNON.Material();


        world.addContactMaterial(new CANNON.ContactMaterial(groundMat, devMat, { 

            friction: 1.0, restitution: 0.0, contactEquationStiffness: 1e8 

        }));

        world.addContactMaterial(new CANNON.ContactMaterial(groundMat, heritageMat, { 

            friction: 1.0, restitution: 0.0, contactEquationStiffness: 1e8 

        }));

        world.addContactMaterial(new CANNON.ContactMaterial(devMat, heritageMat, { 

            friction: 0.95, restitution: 0.0, contactEquationStiffness: 1e8 

        }));

        world.addContactMaterial(new CANNON.ContactMaterial(heritageMat, heritageMat, { 

            friction: 0.95, restitution: 0.0, contactEquationStiffness: 1e8 

        }));


        // --- STATE ---

        let initialHeritage = null;

        let developmentObjects = [];

        let heritageObjects = [];

        const physicsObjects = [];

        const draggableMeshes = [];

        let activeBody = null;


        // --- 3. ENVIRONMENT ---

        scene.add(new THREE.AmbientLight(0xffffff, 0.5));

        const light = new THREE.DirectionalLight(0xffffff, 0.9);

        light.position.set(10, 20, 10);

        light.castShadow = true;

        scene.add(light);


        const boardGeo = new THREE.BoxGeometry(25, 0.5, 25);

        const boardMesh = new THREE.Mesh(boardGeo, new THREE.MeshStandardMaterial({ color: 0xffffff }));

        boardMesh.receiveShadow = true;

        boardMesh.position.y = -0.25;

        scene.add(boardMesh);


        const boardBody = new CANNON.Body({ mass: 0, material: groundMat });

        boardBody.addShape(new CANNON.Box(new CANNON.Vec3(12.5, 0.25, 12.5)));

        boardBody.position.set(0, -0.25, 0);

        world.add(boardBody);


        function createInitialHeritage() {

            const w=3, h=0.8, d=4;

            const mesh = new THREE.Mesh(

                new THREE.BoxGeometry(w,h,d), 

                new THREE.MeshStandardMaterial({ 

                    color: 0x27ae60,

                    emissive: 0x0a3318,

                    emissiveIntensity: 0.15,

                    roughness: 0.8

                })

            );

            mesh.position.set(0, h/2 + 0.25, 0);

            mesh.castShadow = true;

            mesh.userData.isInitial = true;

            

            const body = new CANNON.Body({ mass: 0 });

            body.isDev = false;

            body.material = heritageMat;

            body.addShape(new CANNON.Box(new CANNON.Vec3(w/2, h/2, d/2)));

            body.position.copy(mesh.position);

            

            scene.add(mesh);

            world.add(body);

            initialHeritage = { mesh, body };

        }

        createInitialHeritage();


        // --- OBJECT FUNCTIONS ---

        function addDevelopment() {

            const w=1.8, h=6, d=1.8;

            const mesh = new THREE.Mesh(

                new THREE.BoxGeometry(w,h,d), 

                new THREE.MeshStandardMaterial({ color: 0x666666, metalness: 0.1 })

            );

            spawn(mesh, new CANNON.Box(new CANNON.Vec3(w/2, h/2, d/2)), 15, true);

        }


        function addHeritageBox() {

            const w=1.6, h=1.6, d=1.6;

            const mesh = new THREE.Mesh(

                new THREE.BoxGeometry(w,h,d), 

                new THREE.MeshStandardMaterial({ color: 0x2ecc71, roughness: 0.7 })

            );

            spawn(mesh, new CANNON.Box(new CANNON.Vec3(w/2, h/2, d/2)), 1, false);

        }


        function spawn(mesh, shape, mass, isDev) {

            mesh.position.set((Math.random()-0.5)*10, 12, (Math.random()-0.5)*10);

            mesh.castShadow = true;

            mesh.userData.id = Date.now() + Math.random();

            

            const body = new CANNON.Body({ mass: mass });

            body.isDev = isDev;

            body.material = isDev ? devMat : heritageMat;

            body.linearDamping = 0.95;

            body.angularDamping = 0.98;

            body.userData = { id: mesh.userData.id };

            body.addShape(shape);

            body.position.copy(mesh.position);

            

            body.addEventListener("collide", (e) => {

                if(body.isDev && !e.body.isDev) {

                    triggerAlert("⚠️ Development pressure on heritage!");

                }

            });


            scene.add(mesh);

            world.add(body);

            physicsObjects.push({ mesh, body });

            draggableMeshes.push(mesh);

            if(isDev) developmentObjects.push({ mesh, body });

            else heritageObjects.push({ mesh, body });

        }


        // --- SELECTIVE REMOVAL ---

        function showDevelopmentSelection() {

            if(developmentObjects.length === 0) {

                triggerAlert("No development to remove");

                return;

            }

            showSelection('development', 'Select Development Tower to Remove');

        }


        function showHeritageSelection() {

            if(heritageObjects.length === 0) {

                triggerAlert("No added heritage to remove (central green space is protected)");

                return;

            }

            showSelection('heritage', 'Select Heritage/Greenery to Remove');

        }


        function showSelection(type, title) {

            const modal = document.getElementById('selection-modal');

            const list = document.getElementById('selection-list');

            const modalTitle = document.getElementById('modal-title');

            

            modalTitle.textContent = title;

            list.innerHTML = '';

            

            const objects = type === 'development' ? developmentObjects : heritageObjects;

            

            objects.forEach((obj, index) => {

                const div = document.createElement('div');

                div.className = `object-item ${type}`;

                div.innerHTML = `Object ${index + 1}<br><small>ID: ${obj.body.userData.id.toFixed(0)} | Pos: (${obj.mesh.position.x.toFixed(1)}, ${obj.mesh.position.z.toFixed(1)})</small>`;

                div.onclick = () => removeSpecificObject(obj, type);

                list.appendChild(div);

            });

            

            modal.style.display = 'block';

        }


        function removeSpecificObject(obj, type) {

            removeObject(obj);

            closeSelection();

            triggerAlert(`${type === 'development' ? '🏗️' : '🌳'} Removed selected ${type}`);

        }


        function closeSelection() {

            document.getElementById('selection-modal').style.display = 'none';

        }


        function removeObject(obj) {

            const index = physicsObjects.findIndex(o => o.mesh === obj.mesh);

            if(index > -1) physicsObjects.splice(index, 1);

            const dragIndex = draggableMeshes.indexOf(obj.mesh);

            if(dragIndex > -1) draggableMeshes.splice(dragIndex, 1);

            

            const devIndex = developmentObjects.indexOf(obj);

            if(devIndex > -1) developmentObjects.splice(devIndex, 1);

            

            const herIndex = heritageObjects.indexOf(obj);

            if(herIndex > -1) heritageObjects.splice(herIndex, 1);

            

            scene.remove(obj.mesh);

            world.remove(obj.body);

        }


        function triggerAlert(text) {

            const log = document.getElementById('alert-log');

            const el = document.createElement('div');

            el.className = 'alert-msg';

            el.innerText = text;

            log.appendChild(el);

            setTimeout(() => el.remove(), 4000);

        }


        // --- CONTROLS ---

        const orbit = new THREE.OrbitControls(camera, renderer.domElement);

        orbit.mouseButtons = { RIGHT: THREE.MOUSE.ROTATE };

        

        const drag = new THREE.DragControls(draggableMeshes, camera, renderer.domElement);

        drag.addEventListener('dragstart', (e) => {

            orbit.enabled = false;

            activeBody = physicsObjects.find(o => o.mesh === e.object)?.body;

            if(activeBody) {

                activeBody.mass = 0;

                activeBody.updateMassProperties();

            }

        });

        drag.addEventListener('dragend', (e) => {

            orbit.enabled = true;

            if(activeBody) {

                activeBody.mass = activeBody.isDev ? 15 : 1;

                activeBody.updateMassProperties();

                activeBody.velocity.set(0,0,0);

                activeBody.angularVelocity.set(0,0,0);

                activeBody = null;

            }

        });

        drag.addEventListener('drag', (e) => { 

            if(activeBody) activeBody.position.copy(e.object.position); 

        });


        // Close modal on click outside

        document.getElementById('selection-modal').onclick = (e) => {

            if(e.target.id === 'selection-modal') closeSelection();

        }


        function clearAll() {

            physicsObjects.forEach(o => { scene.remove(o.mesh); world.remove(o.body); });

            developmentObjects = [];

            heritageObjects = [];

            physicsObjects.length = 0; 

            draggableMeshes.length = 0;

            if(initialHeritage) {

                scene.remove(initialHeritage.mesh);

                world.remove(initialHeritage.body);

            }

            createInitialHeritage();

            triggerAlert("🌳 Reset with central heritage/green space");

        }


        // --- LOOP ---

        function animate() {

            requestAnimationFrame(animate);

            world.step(1/60);

            

            physicsObjects.forEach(obj => {

                if(activeBody !== obj.body) {

                    obj.mesh.position.copy(obj.body.position);

                    obj.mesh.quaternion.copy(obj.body.quaternion);

                }

            });


            const devs = developmentObjects.length;

            const her = heritageObjects.length + (initialHeritage ? 1 : 0);

            const equity = her === 0 ? 0 : Math.round((her / (devs + her)) * 100);

            const pressure = Math.min(devs * 3, 30);

            

            document.getElementById('equity-val').innerText = equity + "%";

            document.getElementById('pressure-val').innerText = pressure;

            

            renderer.render(scene, camera);

        }

        animate();

    </script>
